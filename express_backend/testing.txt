// if not using postman for api testing, run these commands in windows powershell for api testing

1) to get all users // get all will not work without token - first login generate token - copy paste token in place of $token
>> Invoke-RestMethod -Method GET -Uri http://localhost:4000/api/users -Headers @{ Authorization = "Bearer $token" }

2) registration
>> Invoke-RestMethod -Method POST -Uri http://localhost:4000/api/users/register -ContentType "application/json" -Body '{"name":"demoUser1","email":"user1@gmail.com","password":"1234","country":"India"}'

3) login 
>> Invoke-RestMethod -Method POST -Uri http://localhost:4000/api/users/login -ContentType "application/json" -Body '{"email":"user1@gmail.com","password":"1234"}'
